## 6 模型 screen200 frozen 运行
```bash
cd "/Users/sam/Documents/Documents - sam的MacBook Pro/论文📑/Distillation/new task"

conda run -n Lion python scripts/ensure_candidate_models.py \
  --labels "Orca-2-7B" "Zephyr-7B-beta" "Qwen2.5-7B-Instruct" "Mistral-7B-Instruct-v0.3" "Yi-1.5-6B-Chat" "ChatGLM3-6B" \
  --local-only \
  --provisional

conda run -n Lion python scripts/run_qualification_v1.py \
  --labels "Orca-2-7B" \
  --model-registry outputs/metadata/provisional/model_registry.labels-orca-2-7b-zephyr-7b-beta-qwen2-5-7b-instruct-mistral-7b-instruct-v0-3-yi-1-5-6b-chat-chatglm3-6b.json \
  --run-root outputs/provisional/screen200_6model_20260407 \
  --report-csv outputs/provisional/screen200_6model_20260407/reports/orca-2-7b.csv \
  --max-new-tokens 256 \
  --resume \
  --screen-only \
  --provisional

conda run -n Lion python scripts/build_screen200_table_v1.py \
  --model-registry outputs/metadata/provisional/model_registry.labels-orca-2-7b-zephyr-7b-beta-qwen2-5-7b-instruct-mistral-7b-instruct-v0-3-yi-1-5-6b-chat-chatglm3-6b.json \
  --labels "Orca-2-7B" "Zephyr-7B-beta" "Qwen2.5-7B-Instruct" "Mistral-7B-Instruct-v0.3" "Yi-1.5-6B-Chat" "ChatGLM3-6B" \
  --run-root outputs/provisional/screen200_6model_20260407 \
  --output-json outputs/provisional/screen200_6model_20260407/screen200_status_table.json \
  --output-md outputs/provisional/screen200_6model_20260407/screen200_status_table.md \
  --max-new-tokens 256
```
